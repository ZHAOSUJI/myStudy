一般使用静态变量来实现递归


    public function queryPath($cid){
        static $path;
        $sql = "select id,parent_id,name from product_category_ali where id = '{$cid}'";
        $info = Yii::$app->db->createCommand($sql)->queryOne();
        if(!$info){
            return false;
        }
        $path[] = $info['name'];
        if($pid = $info['parent_id']) {
            $this->queryPath($pid);
        }
        return $path;
    }
